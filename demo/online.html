Permit.js

Demo Online / Offline

This page demonstrates a basic app prototype where there are different online and offline features. Start by logging in. Now kick them offline!

The Code

HTML
<!-- Login Button -->
<div class="permit permit-none">
  ...
  <button class="permit permit-none permit-issue-online">Login</button>
  ...
</div>

<!-- Logout Button -->
<div class="permit permit-all">
  <button type="button" class="permit-revoke-all">Logout</button>
</div>

<!-- Kick Them Offline -->
<a href="#" class="permit-issue-offline">kick them offline</a>

<!-- Online Mode -->
<div class="permit permit-online">
  online mode content
</div>

<!-- Offline Mode -->
<div class="permit permit-offline">
  offline mode content
</div>
Javascript
$(document).ready(function()
{
    $.permit(
        {permits: ['online','offline']}
    );
});

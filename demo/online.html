Permit.js

Demo Online / Offline

This page demonstrates a basic app prototype where there are different online and offline features. Start by logging in.

The Code

HTML
<!-- Login Button -->
<div class="permit permit-none">
  ...
  <button class="permit permit-none permit-issue-online">Login</button>
  ...
</div>

<!-- Logout Button -->
<div class="permit permit-all">
  <button type="button" class="permit-revoke-all">Logout</button>
</div>

<!-- Offline Mod -->
<a class="btn btn-default permit-issue-offline">Offline</a>

<!-- Online Mode -->
<a class="btn btn-default permit-issue-online">Online</a>


<!-- Offline Mode -->
<div class="permit permit-offline">
  offline mode content
</div>
Javascript
$(document).ready(function()
{
    $.permit(
        {permits: ['online','offline']}
    );
});
